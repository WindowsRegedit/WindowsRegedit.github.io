448. 解冻危机

题目描述
你是一名厨师长，掌管着一家米其林三星餐厅。今天你要制作n只烤鸡，这些鸡都是刚从冰箱拿出来需要解冻。第i只鸡原本含有冰块量为ai份。如果放在厨房自然解冻，每秒钟在一只鸡身上能融化1份冰；如果使用微波炉解冻，每秒钟可以融化k份冰。可惜的是，你只有一个微波炉，微波炉里同时只能放入一只鸡，微波炉运行时间的最小单位是1秒。请问要解冻所有n只鸡，至少需要几秒钟？

输入输出格式
输入格式
输入第一行为正整数n，第二行共n个整数代表每个ai，第三行为正整数k，1<=k<=100,1<=n<=100000，0<=ai<=10000。
输出格式
输出一个整数。

输入输出样例
输入样例#1:
4
2 2 6 6
5
输出样例#1:
2
输入样例#2:
3
2 3 9
5
输出样例#2:
3
输入样例#3:
2
3 6
4
输出样例#3:
3

#include<bits/stdc++.h>
using namespace std;
const int N=100009;
int n,k,a[N];
bool OK(int m){
	int x,cnt=0;
	for(int i=n;i>=1;i--){
		if(a[i]<=m)break;
		x=ceil(1.0*(a[i]-m)/(k-1));
		if((cnt+=x)>m)return 0;
	}
	return 1;
}
int main(){
	cin>>n;
	for(int i=1;i<=n;i++)cin>>a[i];
	cin>>k;
	sort(a+1,a+1+n);
	int l=1;
	int r=a[n];
	int ans=r;
	if(k==1){
		cout<<r<<endl;
		return 0;
	}
	while(l<=r){
		int mid=l+(r-l)/2;
		if(OK(mid))ans=mid,r=mid-1;
		else l=mid+1;
	}
	cout<<ans<<endl;
	return 0;
}

输入输出格式
输入格式
输入文件steps.in 输入正整数n，a和b。1<=n,a,b<=1000。a不等于b。
输出格式
输出文件steps.out 输出一个数。

输入样例#1:
10 2 5

2

10 6 7

#include<bits/stdc++.h>

const int INF=1e9;
int n,a,b,f[N];

	freopen("steps.in", "r", stdin);
	freopen("steps.out", "w", stdout);
	cin>>n>>a>>b;
	f[0]=0;

		f[i]=INF;
		if(i>=a)f[i]=min(f[i],f[i-a]+1);

	if(f[n]==INF)cout<<-1<<endl;
	else cout<<f[n]<<endl;
	return 0;
}

假设你可以选择任意一段连续的工作时间。

输入输出格式
输入格式
输入第一行为正整数n和k，第二行共n个正整数xi，在1到200之间。n<=1000。
输出格式
输出一个正整数代表最小星期数，若无法玩到k个国家输出impossible

输入输出样例

5 3
7 6 7 7 8

输入样例#3:
无
输出样例#3:

#include<bits/stdc++.h>
using namespace std;

int x[1009];
int n,k,l,r,ans;
int cnt[209];
int sum;
int main(){
	cin>>n>>k;
	for(int i=0;i<n;i++)cin>>x[i];
	ans=n+1;
	while(1){
		while(r<n&&sum<k){
			int num=x[r++];
			cnt[num]++;
			if(cnt[num]==1)sum++;
		}
		if(sum<k)break;
		ans=min(ans,r-l);
		int num=x[l++];
		cnt[num]--;
		if(cnt[num]==0)sum--;
	}
	if(ans==n+1)cout<<"impossible";
	else cout<<ans;
	return 0;
}

int ans=0,n=5,f[N][N];

	for(int i=x-1;i>=0;i--)

#include<bits/stdc++.h>
#define N 19
using namespace std;
int dx[4]={0,1,0,-1};
int dy[4]={1,0,-1,0};
int m,n,xR,yR,xJ,yJ,dst[N][N];
char d[N][N];
bool vst[N][N];
struct Node{
	int x,y;
};
int bfs(){
	queue<Node> q;
	vst[xR][yR]=1;
	dst[xR][yR]=0;
	q.push((Node){xR,yR});
	while(!q.empty()){
		Node now=q.front();
		q.pop();
		for(int k=0;k<4;k++){
			int nx=now.x+dx[k],ny=now.y+dy[k];
			if(nx>=1&&nx<=n&&ny>=1&&ny<=m&&d[nx][ny]!='#'&&!vst[nx][ny]){
				vst[nx][ny]=1;
				dst[nx][ny]=dst[now.x][now.y]+1;
				q.push((Node){nx,ny});
				if(nx==xJ&&ny==yJ)return dst[nx][ny];
			}
		}
	}
	return -1;
}
int main(){

547. 关系式矛盾2

题目描述
输入n条简单的关系式，判断是否有可能都成立。关系符号只可能是两种：!=代表不等于关系，==代表等于关系。涉及的字母只可能是a到z的小写字母，每个字母代表一个变量，关系符号左右各有一个字母。

输入输出格式
输入格式
输入第一行为正整数n，n<=100，接着有n条关系式。
输出格式
输出Impossible或Possible

输入输出样例
输入样例#1:
2
a==b
c==d
输出样例#1:
Possible
输入样例#2:
3
x!=y
y==z
z==x

Impossible
输入样例#3:

Impossible

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll N=109;
ll nRelations,m=0,qx[N],qy[N],id[N];
ll root(ll x){
	if(id[x]==x)return x;
	return id[x]=root(id[x]);
}
void unite(ll x,ll y){
	ll rx=root(x),ry=root(y);
	id[rx]=ry;
}
int main(){
	cin>>nRelations;
	for(ll i=0;i<N;i++)id[i]=i;
	for(ll i=0;i<nRelations;i++){
		char u,v,x,y;
		cin>>u>>x>>y>>v;
		if(x=='=')
		    unite(u-'a',v-'a');
		else{
			qx[m]=u-'a';
			qy[m]=v-'a';
			m++;
		}
	}
	for(int i=0;i<m;i++)
	    if(root(qx[i])==root(qy[i])){
	    	cout<<"Impossible";
	    	return 0;
		}
	cout<<"Possible";
	return 0;
}

586. 黄金道

题目描述
你来到了金山，遍地是黄金。在n*m格子的地图里，每格都有高度值和该格子里黄金数量。地图是四向连通的，请你找一条从高到低的下 山通道，使得沿途的黄金总量最大。该黄金道的起点和终点也由你来选择。注意：高度相同的两个格子无法通行

输入输出格式
输入格式
输入的第一行表示区域的行数n和列数m(1<=n,m<=100)。接着是n行m列个整数代表每格高度，高度<=10000。再接着是n行m列个整数代表每格黄金数量，黄金数量<=10000
输出格式
输出沿途的黄金总量最大值。

输入输出样例
输入样例#1:
5 5
1 2 3 4 5
16 17 18 19 6
15 24 25 20 7
14 23 22 21 8
13 12 11 10 9
6 6 6 6 6
6 6 6 6 6
6 6 6 6 6
6 6 6 6 6
6 6 6 6 6
输出样例#1:
150
输入样例#2:
2 3
1 2 3
4 5 6
1 5 9
4 10 1
输出样例#2:
17
输入样例#3:
无
输出样例#3:
无

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll N=109;
ll dx[4]={1,-1,0,0};
ll dy[4]={0,0,-1,1};
ll n,m,f[N][N],h[N][N],mi[N][N],ans;
bool ok[N][N];
ll F(ll x,ll y){
	if(ok[x][y])return f[x][y];
	ok[x][y]=1;
	f[x][y]=mi[x][y];
	for(ll k=0;k<4;k++){
		ll nx=x+dx[k],ny=y+dy[k];
		if(nx>=1&&nx<=n&&ny>=1&&ny<=m&&h[nx][ny]<h[x][y]){
		    f[x][y]=max(f[x][y],F(nx,ny)+mi[x][y]);
		}
	}
	return f[x][y];
}
int main(){
	cin>>n>>m;
	for(ll i=1;i<=n;i++)
	    for(ll j=1;j<=m;j++)
	        cin>>h[i][j];
	for(ll i=1;i<=n;i++)
	    for(ll j=1;j<=m;j++)
	        cin>>mi[i][j];
	for(ll i=1;i<=n;i++)
	    for(ll j=1;j<=m;j++)
	        ans=max(ans,F(i,j));
	cout<<ans<<endl;
	return 0;
}

591. 造物主

题目描述
作为造物主，你要创造出n个物种，编号1到n。其中，i号物种的基因序列由一个长度为5的字符串代表。目前1号物种已经存在，而其他物种会按照你指定的顺序依次从已有的物种衍化而来。某物种从另一物种衍化出来时会消耗你的能量块，恰好等于这两个物种基因序列中总共出现的不同字符的个数。保证不同物种的序列都不同。求造完n个物种至少消耗多少能量块？

输入输出格式
输入格式
输入文件creator.in 输入第一行为正整数n，n<=500。接着共n行每行一个大写字母的字符串，长度均为5。
输出格式
输出文件creator.out 输出一个正整数。

输入输出样例
输入样例#1:
2
ABCDE
FGABC
输出样例#1:
7
说明：ABCDE和FGABC这两个字符串里共有7个不同字符：A,B,C,D,E,F,G
输入样例#2:
3
ABCDE
FGABC
DAXYZ
输出样例#2:
15
输入样例#3:
4
ABCDE
FGABC
DAXYZ
CDEAB
输出样例#3:
20

#include<bits/stdc++.h>
#define N 1000009
using namespace std;
int n,m,id[N];
string gene[N];
struct edge{int u,v,w;};
edge e[N];
int root(int u){
	return id[u] == u?u:id[u]=root(id[u]);
}
bool cmp(const edge&a,const edge&b){
	return a.w<b.w;
}
void Kruskal(){
	sort(e,e+m,cmp);
	for(int u=1;u<=n;u++)id[u]=u;
	int ans=0,cnt=0;
	for(int k=0;k<m;k++){
		int ru=root(e[k].u);
		int rv=root(e[k].v);
		if(ru==rv)continue;
		id[ru]=rv;
		ans+=e[k].w;
		cnt++;
		if(cnt==n-1)break;
	}
	cout<<ans<<endl;
}
int dst(int i,int j){
	set<char>s;
	for(int k=0;k<5;k++){
		s.insert(gene[i][k]);
		s.insert(gene[j][k]);
	}
	return s.size();
}
int main(){
	freopen("creator.in","r",stdin);
	freopen("creator.out","w",stdout);
	cin>>n;
	for(int i=1;i<=n;i++)cin>>gene[i];
	for(int i=1;i<=n-1;i++)
	    for(int j=i+1;j<=n;j++){
	    	e[m].u=i;
	    	e[m].v=j;
	    	e[m].w=dst(i,j);
	    	m++;
		}
	Kruskal();
	return 0;
}
