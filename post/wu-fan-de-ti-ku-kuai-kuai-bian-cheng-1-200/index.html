#include<bits/stdc++.h>
using namespace std;
int main(){
	int n, num[100], maxium=0, minium;
	cin>>n;
	for(int i=0;i<n;i++)cin>>num[i];
	minium=num[0];
	for(int i=0;i<n;i++){
		if(num[i]<minium)minium=num[i];
